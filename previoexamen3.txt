-- Tabla principal: usuarios_procesos (sin necesidad de tabla procesos separada)
CREATE TABLE IF NOT EXISTS usuarios_procesos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    userId INT,
    username VARCHAR(255),
    porcentaje DECIMAL(5,2)
);

-- Tabla de comentadores
CREATE TABLE IF NOT EXISTS comentadores (
    id INT PRIMARY KEY AUTO_INCREMENT,
    id_usuario_proceso INT,
    email VARCHAR(255)
);
